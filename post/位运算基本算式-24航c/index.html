位运算基本算式

如果同学们难以理解或实现位运算的各种复杂构造，建议大家先掌握下面三个位运算，然后结合条件和循环实现更复杂的位运算操作。

1、取值

(n>>i)&1 //提取n的第i位

示例

打印n的补码（int 类型）

#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    for (int i = 31; i >=0; --i) {
        printf("%d",(n>>i)&1);//提取n的第i位
    }
}

打印n的补码（long long 类型）

#include<stdio.h>
int main(){
    long long n;
    scanf("%lld",&n);
    for (int i = 63; i >=0; --i) {
        printf("%lld",(n>>i)&1);//提取n的第i位
    }
}

2、置1

n=n|(1<<i); //对n的第i位置1

示例

对n的第l到r位置1

#include<stdio.h>

int main(){
    int n;
    int l,r;
    scanf("%d",&n);
    scanf("%d%d",&l,&r);
    //对n的第l到r位置1
    for (int i = l; i <=r; ++i) {
        n|=(1<<i);
    }
    printf("%d\n",n);
}

3、置0

n=n&(~(1<<i)); //对n的第i位置0

示例

对n的第l到r位置0

#include<stdio.h>


int main(){
    int n;
    int l,r;
    scanf("%d",&n);
    scanf("%d%d",&l,&r);
    //对n的第l到r位置0
    for (int i = l; i <=r; ++i) {
        n&=~(1<<i);
    }
    printf("%d\n",n);
}

4、打包的函数

int inplace01(int a,int n,int x)  //替换二进制特定位的值,将a的第n位替换成x  //a为要转化的数字，x为替换的值（0或1），n为位数
{//注意返回值和参数类型，要改成相应的类型，而且1ll也要改成相应的类型（unsigned int是1u，long long 是1ll，int是1）

    if (x==0)
        a &= (~(1 << n));   //将a的二进制第n位替换为0
    else  a|=(1<<n);        //将a的二进制第n位替换为1
    return a;
}

int get01(int a,int n)  //提取二进制的特定位，提取a的第n位  //a为提取的数字，n为位数
{//注意返回值和参数类型，要改成相应的类型，而且1ll也要改成相应的类型（unsigned int是1u，long long 是1ll，int是1）
    return    a&(1<<n);     
    //取出a的第n位，返回（若为1，则返回2^n,否则a=0）
}
