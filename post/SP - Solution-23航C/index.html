题目分析

此题通过计算切割前后剩余糖果数量的差值，我们就可以计算出每次切割分出去的糖果数量。

题目规定每个糖果的坐标均为正整数，保证切割糖块的坐标范围不会超过对应 aaa、bbb 、ccc 的范围，因此我们可以得出糖果的剩余数量，即 (a-x_max)*(b-y_max)*(c-z_max) ，其中 x_max 、y_max 、z_max 为切割 xxx 轴、yyy 轴、zzz 轴对应 kkk 的最大值。在示例代码中，为了简化代码， x_max 、y_max 、z_max 使用数组 xx，其值与 opopop 对应。

需要注意的是，糖果剩余数量的表达式计算范围会超过 int 范围，因此需要进行强制类型转换或乘以 1ll .

#include <stdio.h>

int main() {

D 散列查找24

难度 考点
3 散列查找 一维数组

题目分析

根据题目和样例解释，可以得知散列查找（线性探测法）存入数据的基本流程。需要注意的点有：

  • 当 n>mn > mn>m 时，只需要 mmm 次输入即可。

    题目规定，当数组已满，则将接下来要存入的数直接遗弃。因此当 n>mn > mn>m 时，前 mmm 个数据存入数组后，数组存满。之后的数据不会存在数组中。因此我们只需要循环 mmm 次将前 mmm 个数据存入数组后停止输入循环即可。

  • 当 n>mn > mn>m 时，输入循环共循环 mmm 次。当 n<mn < mn<m 时，输入循环共循环 nnn 次。由此可见输入循环的次数是 mmm 与 nnn 的最小值。因此在循环过程中，数组一定没有存满，通过线性查找一定能找到当前输入数据的存入位置。所以无需判定在线性查找过程中是否返回原位置或数组是否存满。

示例代码

#include <stdio.h>

int main() {
    unsigned int hash[1001] = {0};
    int n, m;
    unsigned int x;
    scanf("%d%d", &m, &n);
    int min;
    if (m < n) min = m;
    else min = n;
    for (int i = 0; i < min; ++i) { //以m和n中的最小值作为循环次数
        scanf("%u", &x);
        unsigned int a = x % m;
        while (hash[a] != 0) {
            a++;
            a %= m;
        }
        hash[a] = x;
    }
    for (int i = 0; i < m; ++i) {
        printf("%u ", hash[i]);
    }
}

E CNN24

难度 考点
3 二维数组 循环

题目分析

卷积层的计算需要三个循环。前两层循环用来定位当前计算的元素位置，第三层循环实现当前元素的卷积计算。

计算过程需要注意前两层循环的循环次数（即输出矩阵的形状）为 m−n+1m-n+1m−n+1。

也需要注意卷积计算时要保证输入图像和卷积核的对齐，输出矩阵的第 iii 行第 jjj 列的计算公式为∑k=1n∑l=1nx[k][l]×y[i+k][j+l]\sum_{k=1}^{n}\sum_{l=1}^{n} x[k][l]×y[i+k][j+l]∑k=1n​∑l=1n​x[k][l]×y[i+k][j+l]

示例代码

#include <stdio.h>

int x[100][100];
int y[100][100];
int output[100][100];

int cnn(int m, int n) {
    for (int i = 0; i < m - n + 1; i++)
        for (int j = 0; j < m - n + 1; j++) {
            output[i][j] = 0;//注意初始化
            for (int k = 0; k < n; ++k) {//第i行第j列的输出矩阵卷积计算
                for (int l = 0; l < n; ++l) {
                    output[i][j] += x[k][l] * y[i + k][j + l];
                }
            }
        }
    return m - n + 1;//返回输出矩阵大小
}

int main() {
    int m, n;
    scanf("%d%d", &n, &m);
    for (int i = 0; i < n; i++)//卷积核
        for (int j = 0; j < n; j++)
            scanf("%d", &x[i][j]);
    for (int i = 0; i < m; i++)//输入矩阵
        for (int j = 0; j < m; j++)
            scanf("%d", &y[i][j]);
    int r = cnn(m, n);
    for (int i = 0; i < r; i++) {
        for (int j = 0; j < r; j++)
            printf("%d ", output[i][j]);
        printf("\n");
    }
}

G 超市活动24

难度 考点
4 递归

题目分析

通过题意，我们可以得出长度为 LLL 的活动区域的隔开的顺序数 m[L]=m[L−a]+m[L−b]+m[L−c]m[L]=m[L-a]+m[L-b]+m[L-c]m[L]=m[L−a]+m[L−b]+m[L−c] 。

递归的基本条件是当 L=0L=0L=0 时,隔开的顺序数量为 111 ，即 m[0]=1m[0]=1m[0]=1 ，当 L<0L<0L<0 时,隔开的顺序数量为 000 .

要注意递归过程中要存储已经计算过的顺序数，避免重复计算造成超时。

示例代码

#include <stdio.h>

long long num[409];

long long count(int a, int b, int c, int L) {
    // 如果活动区域的总长小于0，说明没有合法的隔开顺序，返回0
    if (L < 0) {
        return 0;
    }
    // 如果活动区域的总长等于0，说明刚好隔开到最后一个区域，返回1
    if (L == 0) {
        return 1;
    }
    if (num[L] != -1)
        return num[L];
    num[L] = count(a, b, c, L - a) + count(a, b, c, L - b) + count(a, b, c, L - c);
    return num[L];
}

int main() {
    int a, b, c, l;
    while ((scanf("%d%d%d%d", &a, &b, &c, &l)) != EOF) {
        for (int i = 0; i <= l; ++i) {
            num[i] = -1;//初始化计算过的长度的顺序个数为-1，表示未计算过
        }
        printf("%lld\n", count(a, b, c, l));
    }
}

H Gino 的歌单

难度 考点
4 散列表

题目分析

本题涉及到对大量数据进行增删操作，如果尝试使用数组直接模拟这个过程，即使你使用二分查找，也只能使你在查找的过程中有一个 $\mathrm{O}(\log n) $ 的较低的时间复杂度，而插入数据和删除数据都有 O(n)\mathrm{O}(n)O(n) 的时间复杂度，而我们需要做 2×1052 \times 10^52×105 次操作，这显然过于慢了。

本题在输入部分中强调的“按升序给出”其实是在给大家挖坑，诱导大家往二分查找的方向思考。而这题的正确做法是，注意到 xxx 的取值范围比较小，可取值的区间为 −106≤x≤106-10^6 \le x \le 10^6−106≤x≤106，而这就给了我们一个新思路。我们不去直接记录歌单里每个元素的值，而是用一个 cnt 数组，记录可能出现的值都出现过多少次，这样每次新增一个元素 x，就是把 cnt[x] 加一，删去一个元素 x，就是把 cnt[x] 减一，删除全部 x，就是把 cnt[x] 归零。最后输出时，只需根据每个 cnt[x] 的值，将其多次输出即可。不过注意到 xxx 可能为负数，而数组是不可访问其负数位的，因此我们可以以 cnt[x+1000000] 来代替 cnt[x]，把原来 −106∼106-10^6\sim 10^6−106∼106 的值一一映射到数组的第 0∼2×1060\sim 2 \times 10^60∼2×106 位上。最后，每次查找、增加、删除一个元素的时间复杂度来到了 O(1)\mathrm{O}(1)O(1)，最终输出全部元素的过程，其时间复杂度为 O(x)\mathrm{O}(x)O(x)。

#include <stdio.h>
#define plus 1000000
int cnt[2000005];
int main(void)
{
    int n, t, i, x, op;
    scanf("%d%d", &n, &t);
    while (n--)
    {
        scanf("%d", &x);
        cnt[x + plus]++;
    }
    while (t--)
    {
        scanf("%d%d", &op, &x);
        if (op == 1)
        {
            cnt[x + plus]++;
        }
        else if (op == 2)
        {
            if (cnt[x + plus])
            {
                cnt[x + plus]--;
            }
        }
        else
        {
            cnt[x + plus] = 0;
        }
    }
    int sum = 0;
    for (i = 0; i < 2000005; i++)
    {
        sum += cnt[i];
    }
    printf("%d\n", sum);
    for (i = 0; i < 2000005; i++)
    {
        if (cnt[i])
        {
            while (cnt[i]--)
            {
                printf("%d ", i - plus);
            }
        }
    }
    return 0;
}

I 烦躁的邮差

难度 考点
5 dfs

题目分析

本题可以用回溯的思想来做。从数组的第一个位置开始，我们从小到大依次尝试放入 1∼n1\sim n1∼n，然后在对数组的第二个位置进行相同的操作，直到数组的第 nnn 个位置，说明数组其中所有位置都已被填满数字，此时需要输出。而由于我们每一位都是从小到大枚举的，因此最后的输出顺序一定可以按照字典序输出。在枚举过程中，我们需要跳过已经在本位置前被放入过数组的数字，同时为了满足错排条件，还需要跳过等于当前数组下标的数字。

示例代码1

#include <stdio.h>
int arr[15];
int cnt[11]; // 记录每个数字是否出现过
void dfs(int n, int x) // 第一个参数代表最大长度，第二个参数代表当前正在对数组的哪一位进行操作
{
    if (x == n)
    {
        for (int i = 0; i < n; i++)
        {
            printf("%d ", arr[i]);
        }
        printf("\n");
        return;
    }
    for (int i = 1; i <= n; i++)
    {
        if (i - 1 == x) // 不满足错排条件，跳过
            continue;
        if (cnt[i]) // 这个数字已经在之前被使用过了，跳过
            continue;
        cnt[i] = 1; // 标记此数字被使用过了
        arr[x] = i;
        dfs(n, x + 1);
        cnt[i] = 0; // 记得把对此数字的标记取消
    }
}
int main(void)
{
    int n;
    scanf("%d", &n);
    dfs(n, 0);
    return 0;
}

K 矩阵链乘

难度 考点
5 栈、括号匹配

题目分析

本题的关键是解析表达式。注意到题目给出的表达式都有一个特点，即任意两个将要计算的矩阵两边都有括号。根据这个特点，我们可以很自然的想到用栈来模拟括号匹配的过程：遇到字母时将字母入栈；遇到右括号时，则完成一次括号匹配，将栈顶两个字母出栈进行矩阵相乘运算，然后再将结果入栈。因为表达式一定合法，所以无需将括号入栈。

示例代码

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdbool.h>
typedef struct Martrix //建立矩阵结构体
{
    int row;
    int col;
} Martrix;
Martrix m[26], stack[100]; //stack为栈
int top = -1; //栈顶
int main()
{
    int n;
    scanf("%d", &n);
    getchar();
    for (int i = 0; i < n; i++)
    {
        char name;
        scanf("%c", &name);
        int k = name - 'A';
        scanf("%d%d", &m[k].row, &m[k].col); //读入所有矩阵
        getchar(); //消除行尾的'\n'，防止干扰读入矩阵名
    }
    char expr[100]; //读取表达式
    while (scanf("%s", expr) != EOF)
    {
        int len = strlen(expr);
        bool error = false;
        int ans = 0;
        for (int i = 0; i < len; i++)
        {
            if (isalpha(expr[i])) //入栈
                stack[++top] = m[expr[i] - 'A'];
            else if (expr[i] == ')')
            {
                Martrix m2 = stack[top--]; //出栈计算
                Martrix m1 = stack[top--];
                if (m1.col != m2.row)
                {
                    error = true;
                    break;
                }
                ans += m1.row * m1.col * m2.col;
                m1.col = m2.col;
                stack[++top] = m1; //将结果入栈
            }
        }
        if (error)
            printf("error\n");
        else
            printf("%d\n", ans);
    }
}

  1. 归并排序：这个排序方法首推，实际用起来的性能是仅次于快速排序的，特别快，而且具有 O(nlog⁡n)\mathrm{O}(n\log n)O(nlogn) 家族中难以寻求的稳定性（等值条件下相对位置不发生改变）如果你觉得你的归并排序比别的排序慢，那么大概率是你用了 malloc，而 malloc 出来的数组，其效率是远不如普通的数组的。
  2. 堆排序：这个排序方法是 O(nlog⁡n)\mathrm{O}(n\log n)O(nlogn) 家族中最慢的，但是“堆”的思想可不止能够用于排序算法，许多算法题中会用到的“优先队列”就是用“堆”的思想实现的，因此这个排序方法次推
  3. 基数排序：不同于其他排序，这个排序方法时间复杂度是 O(n)\mathrm{O}(n)O(n) 的，更准确的说是 O(nk)\mathrm{O}(nk)O(nk)。因此在大规模数据的排序中（大于 10810^8108），基数排序可以对其他排序形成降维打击。不过其致命缺点就是只能对整型数据排序，对于浮点型数据束手无策。而且在数据总量不太多，但数据范围很广的情况下，用基数排序就显得有些“大炮打蚊子”了。
  4. 希尔排序（壳排序）：这个排序的地位很尴尬，虽然也是八大排序 O(nlog⁡n)\mathrm{O}(n\log n)O(nlogn) 家族的一员，但是基本没他出场的机会，几乎所有场景都有他的上位替代。

}

void build_heap(int *number, int root, int len)
{
    int leaf;
    for (leaf = 2 * root + 1; leaf < len; leaf = 2 * leaf + 1)
    {
        if (leaf + 1 < len)
        {
            if (number[leaf + 1] > number[leaf])
                leaf++;
        }
        if (number[leaf] > number[(leaf - 1) / 2])
        {
            swap(number, (leaf - 1) / 2, leaf);
        }
    }
}
void swap(int *number, int a, int b)
{
    int tmp;
    tmp = number[a];
    number[a] = number[b];
    number[b] = tmp;
}

O Gino 是 DDL 战士 plus

难度 考点
6 排序，递归

题目分析

本题是 Gino 是 DDL 战士的升级版，需要对较长区间进行优先操作。我们虽然没法用递归直接输出操作次序了，但是，我们注意到，对某个点进行操作时，其一定是某个区间的中间（偏右）的元素，并且这个区间的左右端点是固定的！以 n=8n=8n=8 为例，以最左边元素为 111 号位，现在考虑 333 号位对应的区间的左右端点。首先，我们显然知道，n=8n=8n=8 时，其中间偏右的元素为 555 号位，也就是说，555 号位对应的就是 [1,8][1,8][1,8] 这整个区间，其长度为 888。完成 555 号位元素后，可以发现，区间被划分为 [1,4][1,4][1,4] 和 [6,8][6,8][6,8] 两个部分，而 333 号位正好是左区间的中间偏右的元素，因此，333 号位对应的区间就是 [1,4][1,4][1,4]，且只可能是 [1,4][1,4][1,4]，其长度为 444。

于是，我们首先可以通过和之前未加强版问题一样的方法，使用递归，就可以将每个位置的元素所对应的区间的长度记录下来。在这一步之后，我们将每个元素按照其对应的区间的长度由大到小进行排序，而由于同等长度选最右边的规则，在长度相同时按照元素自身编号大小由大到小进行排序，我们就能得到每个元素的操作次序：排在第一位的会第一个被操作，排在第二位的会被第二个操作，以此类推。

注：为什么我们可以这样做？因为本题是优先对较长区间进行操作，而在较长区间被操作之后，它才会生成两个较短区间。也就是说，我们即将对较短区间操作之时，这个较短区间一定是已经形成了的。

也许这样说还不够明白，那么我们换个方向思考：为什么在未加强版中我们不可以这样做？因为那题我们是优先对较短区间进行操作。如果我们这样对每个元素排序，那么排在首位的会是一个对应区间长度为 111 的元素，可是我们可以发现，这个区间根本还没形成呢，我们自然不可能这样对它进行操作。

#include <stdio.h>
#include <stdlib.h>
int arr[1000005];
int len[1000005];    // 编号为i的元素对应的区间的长度
int order[1000005];  // 初始化为0、1、2……n，按照 len[i] 为第一关键字，i为第二关键字排序
void f(int l, int r) // 求每个元素对应的区间的长度
{
    if (l > r)
        return;

    int mid = (l + r) / 2;
    if ((l + r) % 2 == 1)
        mid++;
    len[mid] = r - l + 1;
    f(l, mid - 1);
    f(mid + 1, r);
}
int cmp(const void *a, const void *b)
{
    int a1 = *(int *)a;
    int b1 = *(int *)b;
    if (len[a1] > len[b1])
        return -1;
    else if (len[a1] < len[b1])
        return 1;
    else
        return b1 - a1;
}
int main(void)
{
    int n, i;

    for (i = 0; i < n; i++)
        order[i] = i;

    {
        arr[order[i]] = i + 1;
    }
    for (i = 0; i < n; i++)
    {
        printf("%d ", arr[i]);
    }
    return 0;
}

首先普及一下优先队列相关知识：

普通队列是一种先进先出的数据结构，元素在队列尾部追加，而从队列头部删除。

优先队列则是一种特殊的队列，在这个队列中元素被赋予优先级，当访问元素时，最先出队的元素并不是队列头部元素，而是队列中优先级最高的元素。

再回到本题上，我们可以把元素大小作为一个评判优先级的标准：越小的元素优先级越高。

我们可以从小到大生成各个丑数。最小的丑数是 111 ，而对于任意丑数 xxx，2x、3x2x、3x2x、3x 和 5x5x5x 也都是丑数。这样，就可以用一个优先队列保存所有已经生成的丑数，每次取出最小的丑数，生成 333 个新的丑数。唯一需要注意的是，同一个丑数可能有多种生成方式，所以需要判断一个丑数是否已经生成过。

题目分析

首先，我们容易得知参与嵌套的信封首先长宽需要大于明信片的长宽，需要首先筛选出合适的信封。

其次，能够嵌套的信封序列，其长宽都是升序的，如果我们对信封序列根据长进行排序，最多嵌套的信封序列一定是排序后序列的子序列，且宽是严格升序的。

我们容易想到，可以对信封序列根据长排序后，再求宽的最大上升子序列长度。

但需要注意，在求宽的最大上升子序列时，我们需要保证长相同的信封只能选择一个（题目要求时严格升序），我们可以在按长排序时，再以宽为第二关键字降序排列，保证在求宽的最大上升子序列时，长相同的信封只会选择到一个。

求最大上升子序列长度的 O(nlog⁡n)O(n\log{n})O(nlogn) 算法可以参考这里。

示例代码

#include <stdio.h>
#include <stdlib.h>
#define N 100005
typedef struct node{ // 信封结构体
    int w, h;
} envelope;
envelope env[N];
int LIS[N];
int cmp(const void *aa, const void *bb){ // 排序函数
    envelope* a = (envelope*)aa;
    envelope* b = (envelope*)bb;
    if (a->h < b->h) // 第一关键字升序
        return -1;
    else if (a->h > b->h)
        return 1;
    else{
        if (a->w < b->w) // 第二关键字降序
            return 1;
        else if (a->w > b->w)
            return -1;
        else
            return 0;
    }
}
int binFind(int l, int r, int data){ // 最大上升子序列算法所用的二分函数，用于查找data中第一个大于等于data的位置
    while (l < r) {
        int mid = (l + r) / 2;
        if (LIS[mid] >= data) {
            r = mid;
        }else{
            l = mid + 1;
        }
    }
    return l;
}
int main(){
    int n, W, H;
    scanf("%d", &n);
    scanf("%d %d", &W, &H);

        int w, h;

            env[m].w = w;
            m++;
        }

    m = 0;
    for (int i = 0; i < n; i++) { // 求最大上升子序列长度

            m++;
            continue;
        }
        if (LIS[m-1] < env[i].w) {
            LIS[m] = env[i].w;
            m++;
            continue;
        }
        int loc = binFind(0, m-1, env[i].w);
        LIS[loc] = env[i].w;
    }
    printf("%d", m);
}

        return 1;
    }
