B 算个乘方（水题速来）

此题禁用pow函数

  • pow函数的输出类型是double

  • 当pow函数输出值过大时，误差会超过1。因此整数范围内禁用pow

  • 整数范围的乘方

    long long res=1;
for (int i = 0; i < k; ++i) {
    res*=n;
}
printf("%lld\n",res);

    特别的，2的n次方=1<<n （注意n的范围，当n大于等于32时1改成1ll）

题解

#include <stdio.h>


int main()
{
   long long n,k;

    while (scanf("%lld%lld",&n,&k)!=EOF)
    {
        long long res=1;
        for (int i = 0; i < k; ++i) {
            res*=n;
        }
        printf("%lld\n",res);
    }

}

字符画题搞清楚每一行输出字符数量与行数的关系即可

#include <stdio.h>


int main()
{
   int n;
    scanf("%d",&n);
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j < n-i; ++j) {
            printf(" ");
        }
        for (int j = 0; j < 2*i-1; ++j) {
            printf("A");
        }
        printf("\n");

    }

}

E 开幕式

题解

#include <stdio.h>

int main()
{
    int n;
    int a;
    int b;
    int num=0;
    scanf("%d%d",&n,&a);
    num+=a;
    for (int i = 0; i < n-1; ++i) {
        scanf("%d",&b);
        num+=b;

        while(b!=0)
        {
            int temp=a%b; //因为a和b的值都要更新，所以用temp来保存！
            a=b;
            b=temp;
        }

    }
    printf("%d",num/a);
}
