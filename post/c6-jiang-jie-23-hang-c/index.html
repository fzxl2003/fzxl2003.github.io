关于输入

  • scanf("%s")和gets()
读入前 中止条件 读入后
scanf("%s") 会忽略\n和空格 遇到空格或\n 空格或\n会留在缓冲区内
gets() 不会忽略\n 遇到\n 把\n读入并转换为0作为字符串结束符
  • scanf("%c")与getchar()

在字符读入上完全等价，只是用法和返回值不同。

scanf("%c",&a);
a=getchar()

  • %d等数字读入

    %d、%lld、%lf等数字读入

    读入前 中止条件 读入后
    %d、%lld、%lf 忽略所有空白符 遇到非数字字符 不作处理

    如果在读入前遇到了非数字字符，直接读入失败，不进行读入和赋值

  • 何时需要使用getchar()

    • 当先用%d读入一个数字，然后接着用gets()或getchar()、%c去读下一行的字符或字符串。

    • 此时%d只会读数字，而\n会留下来。

    • 因此需要getchar()去把\n吃掉，方便后续使用gets()或getchar()、%c去读下一行的字符或字符串。

示例代码1

#include<stdio.h>
int main() {
    int n, m;
    char prt[100];
    scanf("%d%d", &n, &m);
    //计算数列前 n 项和
    int sign = 1;
    double sum = 0;
    for (int i = 0; i < n; ++i) {
        sum += 4 * sign * 1.0 / (2 * i + 1);
        sign = -sign;
    }
    //构造输出格式
    sprintf(prt, "f(%d,%d) = %%.%df", n, m, m);//prt="f(100,2) = %.2f"假设n=100，m=2
    //输出结果
    printf(prt, sum);
    return 0;
}

示例代码2

#include<stdio.h>
int main() {
    int n, m;
    char prt[100];
    scanf("%d%d", &n, &m);
    //计算数列前 n 项和
    int sign = 1;
    double sum = 0;
    for (int i = 0; i < n; ++i) {
        sum += 4 * sign * 1.0 / (2 * i + 1);
        sign = -sign;
    }
    //构造输出格式
    sprintf(prt, "f(%%d,%%d) = %%.%df", m);// prt="f(%d,%d) = %.2f"假设n=100，m=2
    //输出结果
    printf(prt, n, m, sum);
    return 0;
}

F 象棋马走日

#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <time.h>
int qipan[105][105];
int main() {
    int m,n,t;
    scanf("%d%d%d",&m,&n,&t);
    int wz[8][2]={{-2,-1},{-2,1},{-1,-2},{-1,2},{1,-2},{1,2},{2,-1},{2,1}};
    while (t--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        x--;y--;
        qipan[x][y]=-1;
        for (int i = 0; i < 8; ++i) {
            if (x+wz[i][0]>=0&&x+wz[i][0]<m&&y+wz[i][1]>=0&&y+wz[i][1]<n)
            {
                if (qipan[x+wz[i][0]][y+wz[i][1]]!=-1)
                   qipan[x+wz[i][0]][y+wz[i][1]]=1;
            }
        }
    }
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (qipan[i][j]==1 || qipan[i][j]==-1)
                printf("0");
            else
                printf("1");
        }
        printf("\n");
    }
}
