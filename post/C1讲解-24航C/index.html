  • \n 换行符 &&&&&&&

4.printf()常用字段类型说明符

  • %d： － 以十进制方式输出带标记整数（正数默认不输出标记）； int 十进制 //////////////////////////////
  • %u： － 以十进制方式输出无标记整数； unsigned int 十进制 /////////////////////
  • %x — 以十六进制方式输出整数 int 十六进制
  • %o： － 以八进制方式输出整数 int 八进制
  • %lf： － 以小数方式输出单、双精度实数； double 小数方式 /////////////////////
  • %e： － 以指数方式输出单、双精度实数； double 科学计数法
  • %g： － 以%f，%e中较短的输出宽度输出单、双精度实数； double
  • %c： － 输出单个字符； char ////////////////////////////////////
  • %s： － 输出字符串。 char* ///////////////////////
  • %lld -以十进制方式输入长整型 （linux系统） long long ///////////////////////////
  • %llu 以十进制方式输出unsigned long long unsigned long long /////////////////////////

1、语法

else if (<表达式>) <语句>
else if (<表达式>) <语句>
......
else <语句>

1、<表达式>既可以是逻辑表达式也可以是数值

if (a==1){}//a为1时执行{}内语句，反之不执行
if (1){}//永远执行{}内语句
if (0){}//永远不执行{}内语句
if (a){}//a为0时不执行{}内语句，反之执行

2、<语句>含有多个代码时，必须加{}

int a=0;
if (a==1)
	printf("1");
	printf("0");

3、<语句>含有多个代码时，可以不加{}

4、常见的逻辑运算符

< 小于   <= 小于等于  > 大于  >= 大于等于
== 等于  != 不等于

!     一元逻辑非运算，把0值变为1，非0值变成0
&&    逻辑与，两端表达式都为非0值时产生1，否则产生0      and
||    逻辑或，两端表达式中有1个为非0值时产生1，否则产生0  or

if (a==1)
if (a<1)
if (a>1&&b<1)
if (a>1||b<1)
if (!(a>1))

while循环

1、语法

while (<表达式>)  
	<语句>
  • <表达式>为循环条件，条件成立时进入循环（方式与if完全一致）
  • <语句>含有多个代码时，必须加{}
int t, a, b;
scanf("%d", &t);
int i=0;
while(i < t) // while循环t次，处理t组数
{
    // your code
    i=i+1;
}

for循环

1、语法

for(<表达式1>;<表达式2>;<表达式3>) 
		<语句>
  • <表达式1>在循环前执行一次，用于初始化变量
  • <表达式2>为循环条件
  • <表达式3>每次循环结束执行，一般用于i++
  • <表达式1><表达式2><表达式3>均可省略，但分号不能省略.表达式中有多个语句的用逗号隔开
  • 省略<表达式2>则循环始终进行，除非遇到break
int i, t, a, b;
scanf("%d", &t);
for(i=0; i<t; i=i+1) // for循环t次，处理t组数
{
    // your code
}

题解

int main() {
    int n;
    scanf("%d",&n);
    while (n--) {
        int op,a;

        if (op==0) {
            for (int i = 0; i < a; ++i) {
                printf("*");
            }
        }
        else {
            for (int i = 0; i < a; ++i) {
                printf("( )");
            }
        }
        printf("\n");
    }

}
