问题1：数据类型搞不清楚

1、int+int爆int问题

int a=1999999999;
int b=1999999999;
printf("%d\n",a+b);  //错
printf("%lld\n",a+b);  //错
printf("%lld\n",1ll*a+b);  //对
printf("%lld\n",(long long)a+b);  //对
printf("%lld\n",(long long)(a+b);  //错

2、整除问题

除的时候用整除

%lf/%f(输入/输出)                                    double 小数方式 
%c： － 输出单个字符；                                 char  
%lld  -以十进制方式输入长整型                           long long

问题10：滥开不定长数组

#include <stdio.h>

int main() {
    int n;
    scanf("%d",&n);
    int a[n];//错
    //定义数组时，[]只能填常数
}

错误原因：本身C99标准支持不定长数组，因此在本地运行时，使用不定长数组可以正常运行，是因为本地测试n的数值通常很小。但OJ测试时，n的值可能很大，直接造成爆栈，从而错误。

问题11：位运算使用极其复杂的构造，企图一个式子出结果

不好的示例

#include <stdio.h>

int main() {
	int n, op;
	scanf("%d", &n);
	unsigned int a, l, x;
	for (int i = 0; i < n; i++) {
		scanf("%d", &op);
		if (op == 1) {
			scanf("%u", &a);
			x = 0;
			for (int m = 31; m >= 0; m--) {
				if (((a >> m) & 1) == 1) {
					x += 1;
				}
			}
			printf("%u\n", x);
		}
		if (op == 2) {
			scanf("%u%u", &a, &l);
			x = (((a >> (32 - l)) ^ 1) << (32 - l)) + ((a << l) >> l);
			printf("%u\n", x);
		}
		if (op == 3) {
			scanf("%u%u", &a, &l);
			x = (((a >> (32 - l)) & 0) << (32 - l)) + ((a << l) >> l) + ((a >> (33 - l)) << (33 - l));
			printf("%u\n", x);
		}
		if (op == 4) {
			scanf("%u%u", &a, &l);
			x = (((a >> (32 - l)) | 1) << (32 - l)) + ((a << l) >> l);
			printf("%u\n", x);
		}
		if (op == 5) {
			scanf("%u%u", &a, &l);
			x = (a >> (32 - l)) & 1;
			printf("%u\n", x);
		}
	}

	return 0;
}

建议只用取值、置1、置0三种简单的位运算再加各种循环和条件实现复杂的位运算
