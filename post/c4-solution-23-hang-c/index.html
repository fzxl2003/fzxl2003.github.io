A Firefly小姐的字符统计

难度 考点
1 ctype.h 库函数的使用

题目分析

使用多组数据输入的读入方式依次读入字符，使用 ctype.h 中函数判断字符类型，再使对应的统计变量增加即可。

示例代码

#include <stdio.h>
#include <ctype.h> // 头文件

int main(){
    char c;
    int upper = 0, lower = 0, alpha = 0, digit = 0, graph = 0; // 统计变量初始化
    while (scanf("%c", &c) !=EOF) {
        if (isupper(c)) { // 大写字母
            upper++;
        }
        if (islower(c)) { // 小写字母
            lower++;
        }
        if (isalpha(c)) { // 字母
            alpha++;
        }
        if (isdigit(c)) { // 数字
            digit++;
        }
        if (isgraph(c)) { // 可见字符
            graph++;
        }
    }
    printf("%d\n", upper);
    printf("%d\n", lower);
    printf("%d\n", alpha);
    printf("%d\n", digit);
    printf("%d", graph);
    return 0;
}

B Fourier Series Expansion

难度 考点
1 math.h，循环

题目分析

使用 math.h 中的函数计算公式 $a_k\cos{kx_0} $ 和 $ b_k\sin{kx_0}$， 使用循环累加即可。

示例代码

#include <stdio.h>
#include <math.h> //头文件

int main() {
    double x;
    int n;
    scanf("%lf%d", &x, &n);
    double fx = 0.0; // 求和变量初始化
    for (int k = 0; k <= n; k++) {
        double a, b;
        scanf("%lf%lf", &a, &b);
        fx += (a * cos(k * x) + b * sin(k * x)); // 累加
    }
    printf("%.2f", fx); // 注意输出小数位数
}

D(n)={0, n=11, n=2(n−1)[D(n−1)+D(n−2)], n≥3D(n)= \begin{cases} 0 & \text{, } n=1 \\ 1 & \text{, } n=2 \\ (n-1)[D(n-1)+D(n-2)] & \text{, } n \ge 3 \end{cases} D(n)=⎩⎪⎨⎪⎧​01(n−1)[D(n−1)+D(n−2)]​, n=1, n=2, n≥3​

可以发现，n≥3n\ge 3n≥3 的表达式是我们进行递归的主要使用的表达式，而 n=1n=1n=1 和 n=2n=2n=2 是递归的终止条件。递归一定要有终止条件，否则就会“只递不归”，使函数不断调用自身直到程序崩溃。

此外，注意到这个递推关系式最后得到的值可能很大，因此需要使用 long long 型变量来计算 D(n)D(n)D(n) 的值。

示例代码

#include <stdio.h>
long long f(int n)
{
    if (n == 1) // 递归终止条件
        return 0;
    else if (n == 2) // 递归终止条件
        return 1;
    else
        return (n - 1) * (f(n - 1) + f(n - 2));
}
int main()
{
    int n;
    while (scanf("%d", &n) != EOF)
    {
        printf("%lld\n", f(n));
    }
    return 0;
}

F czx 的 S 数列

难度 知识点
3~4 函数，递归

题目分析

直接按照题目要求实现相应的递归函数即可。

对于递归函数，其实只需要关注它的 终止条件 即可，终止条件 就是递归的 出口。设置好合适的终止条件后，便可以把这个递归函数当作一个“黑箱”来使用，并在函数中自己向下调用自己。

补充

另外，这道题目实际上是要求打印出移动 NNN 阶汉诺塔的步骤顺序，其中每一步可以用整数表示，整体组成了一个特定的序列 SNS_NSN​。

从题目描述中可以看出，SNS_NSN​ 是通过将 SN−1S_{N-1}SN−1​、NNN、SN−1S_{N-1}SN−1​ 这三部分连接而成的，也就是说，SNS_NSN​ 的构成方式可以看作是在 SN−1S_{N-1}SN−1​ 的两侧加上 NNN 这个数字。这也符合汉诺塔问题的解法：将 NNN 阶汉诺塔从 AAA 经过 BBB 移动到 CCC 可以分解为将 N−1N-1N−1 阶汉诺塔从 AAA 经过 CCC 移动到 BBB，然后将第 NNN 个盘子从 AAA 直接移动到 CCC，最后将 N−1N-1N−1 阶汉诺塔从 BBB 经过 AAA 移动到 CCC。

因此，SNS_NSN​ 就是按照这个规则不断递归构造而成的。在打印 SNS_NSN​ 的过程中，我们会不断地将 SN−1S_{N-1}SN−1​ 的内容复制两次，在中间插入 NNN，这样就得到了 SNS_NSN​。

示例代码 1

#include <stdio.h>
void print(int x) {
    if (x == 1) {
        printf("1");
        return;
    }
    print(x - 1);
    printf(" %d ", x);
    print(x - 1);
}
int main() {
    int n;
    scanf("%d", &n);
    print(n);
    return 0;
}

通过观察规律，可以发现 SNS_NSN​ 一共有 2N−12^N-12N−1 个数。

设第 kkk 个数为 sks_ksk​。正整数 kkk 的二进制表示中，最低为 111 的位是第 xxx 位（设最低位为第 000 位），则 sk=x+1s_k = x+1sk​=x+1。

例如 s6=2s_6 = 2s6​=2，666 的二进制表示为 (110)2(110)_2(110)2​，最低为 111 的位是第 111 位。

该规律可严格证明（可以使用数学归纳法等）。

据此，我们可以使用循环+位运算的方式解决本题。

本思路是“自底而上”的编程思路，而递归则一般是“自顶而下”的编程思路。一般而言，自顶而下的算法会更好设计。

#include <stdio.h>
int main()
{
	int n;
	scanf("%d", &n);
	for(int i = 1; i < 1 << n; ++i)
	{
		int k = 0;
		while((i >> k & 1) == 0) ++k;
		printf("%d ", k + 1);
	}
	return 0;
}

G 林克的修炼

难度 考点
4~5 记忆化递归

题目分析

Hint

用函数 f(k)f(k)f(k) 表示“从 kkk 号神庙进入，林克最终的能力值”，则不难看出，f(k)=ak+f(bk)f(k) = a_k+f(b_k)f(k)=ak​+f(bk​)。

利用递归函数解决问题吧~

可以在递归过程中把计算结果存储到数组 F[k]F[k]F[k] 中，下次如果再需要，则无需耗费时间计算 f(k)f(k)f(k)，直接使用 F[k]F[k]F[k]​ 的值即可。

在Hint中基本已经把本题思路完全给出。

可以设 f(0)=0f(0)=0f(0)=0 作为递归的基本情况。

由于只要进入修炼场的某个神庙，最终能力值就一定会大于 000，因此可以用 F[k]=0F[k]=0F[k]=0 表示未计算过 f(k)f(k)f(k)。

由于 104×105=109<2×10910^4\times 10^5=10^9<2\times 10^9104×105=109<2×109​，因此本题使用 int 类型进行计算即可，无需使用 long long。

示例代码

#include <stdio.h>
int a[100005], b[100005], F[100005];
int f(int k)
{
	if(!k) return 0;      //递归基本情况：若k为0，表示离开神庙
	if(F[k]) return F[k]; //若F[k]不为0，说明计算过f(k)无需再次计算，直接返回F[k]
	return F[k] = a[k] + f(b[k]);
	/* 上面这条语句等价于：
	F[k] = a[k] + f(b[k]);
	return F[k];
	*/
}
int main()
{
	int n;
	scanf("%d", &n);
	for(int i = 1; i <= n; ++i)
		scanf("%d", &a[i]);
	for(int i = 1; i <= n; ++i)
		scanf("%d", &b[i]);
	int ans = 0;
	for(int i = 1; i <= n; ++i)
		if(f(i) > ans) ans = F[i];
	/* 也可以写作：
	{
		int t = f(i);
		if(t > ans) ans = t;
	}
	*/
	printf("%d", ans);
	return 0;
}

补充

通过记忆化递归（使用数组存储递归计算结果）的方式，可以有效降低时间复杂度。

本题而言，直接递归最坏时间复杂度为 O(n2)O(n^2)O(n2)，记忆化递归的时间复杂度为 O(n)O(n)O(n)​。

H 陈氏定理

难度 考点
4~5 判断素数，分解质因数，循环/分支结构

题目分析

基本思路是枚举法，枚举 aaa，进而判断 aaa 和 b=n−ab=n-ab=n−a 是否为素数或半素数。然后按照输出要求输出即可。

因此，我们需要构造能够判断一个数是否为素数或半素数的函数。

思路 1

书上例题中就有判断是否为素数的函数，如下方代码：

int is_prime(int num) //判断num是否为素数
{
	if (num < 2) return 0;
	for (int i = 2; i * i <= num; i++) //枚举至根号num即可
		if (num % i == 0) return 0;
	return 1;
}

我们可以利用该函数写出判断是否为半素数的函数：

int is_semiprime(int num) //判断num是否为半素数
{
	for (int i = 2; i * i <= num; i++)
	{
		if(num % i == 0) //i为num的约数，即num = i * (num / i)
		{
			if(is_prime(i) && is_prime(num / i)) //若num可以分解为两个素数相乘，说明是半素数
				return 1;
		}
	}
	return 0;
}

进而利用这两个函数，就可以通过枚举法去求出所有满足条件的算式。具体代码如下：

思路 2

我们可以构造一个更通用的函数：返回输入参数 xxx 的素因子个数。如下：

int prime_factors_num(int x) //返回x的素因子个数
{
	int cnt = 0;
	for(int i = 2; i * i <= x; i++) //注意循环条件
	{
		while(x % i == 0)
		{
			x /= i;
			cnt++;
		}
	}
	if(x != 1) cnt++; //这条语句一定不能丢
	return cnt;
}

若 xxx 为素数，等价于 xxx 恰好有 111 个素因子；若 xxx 为半素数，等价于 xxx 恰好有 222​ 个素因子。

该算法比思路1的方法更通用且效率更高。

I Gino 是 DDL 战士

难度 考点
5 递归

题目分析

本题要求每次找到最短连续未完成作业区间进行操作，而这可以引起我们的注意。一段较长区间被从中间划分为两个较短区间后，不论选取哪边进行下一步操作，之后的每一步操作，总的最短区间一定产生于新的被划分出来的两个区间中。更具体的，假设长度为 nnn （nnn 为偶数）的区间被划分为左边的长度为 n2\cfrac{n}{2}2n​ 的区间，和右边长度为 n2−1\cfrac{n}{2} - 12n​−1 的区间，那么直到右边区间被完全完成之前，左边的区间都一定会保持初始状态，因为后续对右边区间进行操作而产生的小区间一定会比左边区间要短。

这样的特性就给我们带来了递归的思路。完成对长度为 nnn 的区间的操作，可以分为三步：完成中间（靠右）的元素，完成左（右）区间，完成右（左）区间。

一个很直观的思路就是使用数组来模拟整个过程，可是本题有空间限制，使用数组来模拟最多可以得到 MLE 0.6 的成绩。

既然不让使用数组，那么就只能在递归的过程中直接输出了，这就又有一个问题，输出必须从左向右输出，而本题有时候会先对左区间操作，有时候会先对右区间操作，这可怎么办呢？

解决方法也很简单，只需要在递归的时候再多传入一个参数，告诉这次操作是第几次进行的就行了。设区间长度为 nnn，操作次序是 iii，即这段区间的处于中间（靠右）的元素是第 iii 个被完成的，我们可以推出如下结论：

  • 若 nnn 为奇数，划分出来的两个区间长度都是 n−12\cfrac{n-1}{2}2n−1​，因此中间元素被操作后，先对左区间进行操作，它的操作次序是 i+1i+1i+1。再对右区间进行操作，由于它是在中间元素和左区间所有元素被完成之后才开始操作的，因此它的操作次序是 i+1+n−12i+1+\cfrac{n-1}{2}i+1+2n−1​。最后按照从左向右的输出顺序，把对于左区间的操作放最前面，对中间元素的操作放中间，对右区间的操作放最后。

  • 若 nnn 为偶数，划分出来的左区间长度为 n2\cfrac{n}{2}2n​，右区间长度为 n2−1\cfrac{n}{2} - 12n​−1。因此中间靠右元素被操作后，先对右区间进行操作，它的操作次序是 i+1i+1i+1。再对左区间进行操作，由于它是在中间元素和右区间所有元素被完成之后才开始操作的，因此它的操作次序是 i+n2i+\cfrac{n}{2}i+2n​。最后按照从左向右的输出顺序，把对于左区间的操作放最前面，对中间靠右元素的操作放中间，对右区间的操作放最后。

  • 最后是终止条件：当区间长度 n=0n=0n=0 时，它就没有任何元素进行操作了，此时需要 return 终止递归。

示例代码

#include <stdio.h>
void f(int n, int order)
{
    if (n == 0)
        return;
    if (n % 2 == 1)
    {
        f(n / 2, order + 1);
        printf("%d ", order);
        f(n / 2, order + 1 + n / 2);
    }
    else
    {
        f(n / 2, order + n / 2);
        printf("%d ", order);
        f(n / 2 - 1, order + 1);
    }
}
int main(void)
{
    int n;
    scanf("%d", &n);
    f(n, 1); // 对长度为n的区间进行操作，其中间元素的操作次序是1
    return 0;
}

难度 考点

题目解析

首先用一个数组a[i]的第x位是否是111来记录xxx是否是iii的前提条件。然后用递归的方法一个一个完成任务，最终输出结果，详见代码注释。

示例代码

#include <stdio.h>
#include <stdlib.h>
#define MAXN 1+30

int completed[MAXN];
int visiting[MAXN];
int n, t;
int time[MAXN];
int map[MAXN];

int find(int s) {//返回值：任务s是否可被完成
	int flag = 1;
	if (completed[s] == 1) {//已完成，直接返回
		return 1;
	} else {

		visiting[s] = 1;
		for (int i = 1; i <= n; ++i) {
			if (((map[s] >> i) & 1) == 1) {//提取任务s的前提信息
				if (visiting[i] == 0) {
					flag = find(i);
				} else {//访问到了当前正在访问的任务，说明有环
					return 0;
				}
			}
		}
		completed[s] = 1;//任务完成
		visiting[s] = 0;
		return flag;
	}

}

int main() {

	scanf("%d%d", &n, &t);
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &time[i]);
		map[i] = 0;//map数组记录任务i的前提条件们
		completed[i] = 0;//completed数组记录任务i是否已被完成
		visiting[i] = 0;//visiting数组记录任务i是否正被访问
	}
	for (int i = 1; i <= n; ++i) {
		int k;
		scanf("%d", &k);
		for (int j = 1; j <= k; ++j) {
			int x;
			scanf("%d", &x);
			map[i] |= (1 << x);//用int类型变量的第x位是否是1来记录x是否是i的前提
		}
	}
	if (find(t)) {
		int ans = 0;
		for (int i = 1; i <= n; ++i) {
			ans += completed[i] * time[i];
		}
		printf("%d\n", ans);
	} else {
		printf("Dead Lock!!\n");
	}

	return 0;
}
