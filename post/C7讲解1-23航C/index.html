#include <stdio.h>
typedef struct vec
{
    int x, y, z;
} vec;
vec cross( vec u,  vec v)
{
    vec s ;
    s.x=u.y * v.z - u.z * v.y;
    s.y=u.z * v.x - u.x * v.z;
    s.z= u.x * v.y - u.y * v.x;

    return s;
}
void print_vec( vec a)
{
    printf("%d %d %d\n", a.x, a.y, a.z);
}
int main()
{
    vec a[6];
    for (int i = 1; i <= 5; i++)
    {
        scanf("%d%d%d", &a[i].x, &a[i].y, &a[i].z);
    }
    print_vec(cross(a[1], a[3]));
    print_vec(cross(a[3], a[5]));
    print_vec(cross(a[4], a[1]));
    print_vec(cross(a[2], a[5]));
    print_vec(cross(a[3], a[1]));
    return 0;
}

#include <stdio.h>

int cmp(const void *p1, const void *p2) {  //升序排列
    int *a = (int *) p1, *b = (int *) p2; //这里排列的数据类型是int，如果要排列其他类型，需要把该行的int改成相应的类型
    if (*a < *b)return -1;
    else if (*a > *b) return 1;
    else return 0;
}

int cmp1(const void *p1, const void *p2) {  //降序排列
    int *a = (int *) p1, *b = (int *) p2;//这里排列的数据类型是int，如果要排列其他类型，需要把该行的int改成相应的类型
    if (*a > *b)return -1;
    else if (*a < *b) return 1;
    else return 0;
}
