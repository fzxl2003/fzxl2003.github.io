质数判断

判断num是不是质数，如果是质数返回1，如果不是质数返回0

int iszhi(int num)
{
    if (num==1)
    {
        return 0;
    }
    if (num==2)
    {
        return 1;
    }
    for (int i = 2; i <= sqrt(num); ++i) {
        if (num%i==0)
        {
            return 0;
        }
    }
    return 1;

}
