C 小蓝蓝吃蛋糕

#include <stdio.h>
#include <math.h>
int main(){
    int n,t,s;
    scanf("%d%d%d",&n,&t,&s);
    if (t==0)
    {
        printf("0");
        return 0;
    }
    if (s%t==0)
    {
        n-=(s/t);
    }
    else
        n-=(s/t+1);
    if (n<0)
        printf("0");
    else
        printf("%d",n);
}

E 伪随机码生成

#include <stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    int a[10]={0};
    for (int i = 0; i < 4; ++i) {
        scanf("%d",&a[i]);
    }
    while (t--)
    {
        int f=a[1]^a[3];
        printf("%d",a[3]);
        a[3]=a[2];
        a[2]=a[1];
        a[1]=a[0];
        a[0]=f;
    }
}

#include <stdio.h>
#include <math.h>

int main(){
    int t;
    scanf("%d",&t);
    int a=0,b;
    for (int i = 0; i < 4; ++i) {
        scanf("%d",&b);
        a|=(b<<i);
    }
    while (t--)
    {
        int f=((a>>1)&1)^((a>>3)&1);
        printf("%d",(a>>3)&1);
        a<<=1;
        a|=f;
    }
}

//-1 正在亮，但马上熄灭的
//2 没亮，但马上亮的
//-2  亮过且熄灭的
#include<stdio.h>
int main() {
    int n;
    int s[105]={0};
    int sit[105]={0};
    int k;
    scanf("%d",&n);
    for (int i = 1; i <= n; ++i) {
        scanf("%d",&s[i]);
    }
    scanf("%d",&k);
    int liang_num=1;
    sit[k]=1;
    s[k]++;
    while (liang_num!=0)
    {
        for (int i = 1; i <=n ; ++i) {
            if (sit[i]==1)
            {
                s[i]--;
                if (s[i]==0)
                {
                    sit[i]=-1;
                    liang_num--;
                }
                else
                    printf("%d ",i);
            }
            if (sit[i]==0)
            {
                if (sit[i-1]==1 ||sit[i-1]==-1||sit[i+1]==1)
                {
                    if (s[i]!=0)
                    {
                        sit[i]=2;
                        liang_num++;
                        printf("%d ",i);
                    }


                }
            }
        }
        for (int i = 1; i <=n ; ++i) {
            if (sit[i]==-1)
                sit[i]=-2;
            if (sit[i]==2)
                sit[i]=1;
        }
        printf("\n");
    }
}
