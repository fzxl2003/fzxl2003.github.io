B 寻找字符

难度 考点
2 指针，字符串函数

题目分析

利用 strchr 和 strrchr 函数搜索字符串 sss 中字符 ccc 第一次出现和最后一次出现的位置。代码如下：

char *p = strchr(s, c)；
char *q = strrchr(s, c);

根据指针 p, q 的值进行判断：

  • 若 ccc 在 sss 中未出现，则 p, q 均为空；
  • 若 ccc 在 sss 中恰好出现一次，则 p, q 相等，均指向 ccc 在 sss 中唯一出现的位置，p - s 的值就是答案；
  • 若 ccc 在 sss 中出现不止一次，则 p, q 分别指向 ccc 在 sss 中第一次出现的位置和最后一次出现的位置，q - p 就是答案。

分类讨论即可。

示例代码

#include <stdio.h>
#include <string.h>
char s[100005], c;
int main()
{
	gets(s);
	while(~(c = getchar()))
	{
		char *p = strchr(s, c), *q = strrchr(s, c);
		if(p == NULL)
			printf("-1\n");
		else if(p == q)
			printf("%d\n", p - s);
		else
			printf("%d\n", q - p);
	}
	return 0;
}

C 小宇学矩阵乘法

难度 考点
2 二维数组，循环

题目分析

根据线性代数所学的知识，对于 mmm 行 nnn 列的矩阵 AAA 和 nnn 行 kkk 列的矩阵 BBB

A=(a11⋯a1n⋮⋱⋮am1⋯amn), B=(b11⋯b1k⋮⋱⋮bn1⋯bnk)A=\begin{pmatrix} a_{11} & \cdots & a_{1n} \\ \vdots & \ddots & \vdots \\ a_{m1} & \cdots & a_{mn} \end{pmatrix} ,\,B=\begin{pmatrix} b_{11} & \cdots & b_{1k} \\ \vdots & \ddots & \vdots \\ b_{n1} & \cdots & b_{nk} \end{pmatrix} A=⎝⎜⎛​a11​⋮am1​​⋯⋱⋯​a1n​⋮amn​​⎠⎟⎞​,B=⎝⎜⎛​b11​⋮bn1​​⋯⋱⋯​b1k​⋮bnk​​⎠⎟⎞​

设 A×BA\times BA×B 的结果为矩阵 $C=\begin{pmatrix}
c_{11} & \cdots & c_{1k} \
\vdots & \ddots & \vdots \
c_{m1} & \cdots & c_{mk}
\end{pmatrix} $

其中，cij=∑l=1nailblj (i=1,⋯ ,m, j=1,⋯ ,k)c_{ij}=\sum\limits_{l=1}^{n} a_{il}b_{lj}\,(i=1,\cdots,m,\,j=1,\cdots,k)cij​=l=1∑n​ail​blj​(i=1,⋯,m,j=1,⋯,k)。

按公式循环计算即可。需要注意的是计算结果可能会超出 int 范围，需要保证计算范围和输出数组都需要在 long long 范围内。

示例代码

#include <stdio.h>
int a[50][50];
int b[50][50];
long long c[50][50];
int main()
{
	int m, n, k;
	scanf("%d%d%d", &m, &n, &k);
	// 读入A
	for (int i = 0; i < m; ++i)
		for (int j = 0; j < n; ++j)
			scanf("%d", &a[i][j]);
	// 读入B
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < k; ++j)
			scanf("%d", &b[i][j]);

	// 计算C
	for (int i = 0; i < m; ++i)
		for (int j = 0; j < k; ++j)
			//计算cij
			for (int l = 0; l < n; ++l)
				c[i][j] += (1ll * a[i][l] * b[l][j]); //注意计算范围是long long范围
	// 输出C
	for (int i = 0; i < m; ++i)
	{
		for (int j = 0; j < k; ++j)
		{
			printf("%lld ", c[i][j]);
		}
		printf("\n");
	}
	return 0;
}

E 诡秘符文

难度 考点
4 字符串处理

题目分析

本题主要考察大家对于 strstr, strcpy, strcat 等字符串函数的调用。

将 sss 串中的 s1s1s1 串都转为 s2s2s2 串，如果仅进行一次操作，可以考虑参考课件，将应当输出 s1s1s1 的地方跳过，转而输出 s2s2s2 。

多次对其进行操作，则可以参考以上思路，将输出这一行为变化为输出到一个临时字符串 $ temp$ 中，操作完毕后再把临时字符串 $ temp$ 赋还给 sss 。

本题示例代码中采用 strncpystrncpystrncpy 和 strcatstrcatstrcat 简化输出到临时字符串这一行为，但逐字符输出也是可行的。

示例代码 1

#include <stdio.h>
#include <string.h>
char s[10005];
char s1[10005];
char s2[10005];
char temp[10005]; //临时存储处理字符
char t1[10005]; //存储截取字符串
int main()
{
	int n;
	scanf("%d", &n);
	getchar(); //读取整数后面的换行
	gets(s); //s
	while (n--)
	{
		gets(s1); //s1
		gets(s2); //s2
		char* p1 = s; //开始查找位置
		char* p2 = 0; //s中s1的位置

		while ( (p2 = strstr(p1, s1)) != NULL ) //当s中还有s1
		{
			int l = p2 - p1; //上一个s1之后到下一个s1开始之前的长度
			strncpy(t1, p1, l); //将两个s1之前的字符串截取
			t1[l] = '\0'; //注意strncpy需要补零
			strcat(temp, t1); //拼接到temp中
			strcat(temp, s2); //拼接s2
			p1 = p2 + strlen(s1); //跳转到查找的s1之后
		}
		strcat(temp, p1); //拼接最后一个查找到的s1之后的字符串
		strcpy(s, temp); //将temp赋给s
		memset(s1, 0, 10005); //清空字符串
		memset(s2, 0, 10005);
		memset(temp, 0, 10005);
		memset(t1, 0, 10005);
	}
	puts(s);
	return 0;
}

F ExGCD

难度 考点
4 指针，递归

题目分析

按照伪代码和Hint写出递归函数即可。

示例代码 1

#include<stdio.h>
void ExGCD(int a, int b, int *d, int *x, int *y)
{
    if(b == 0)
    {
        *d = a, *x = 1, *y = 0;
        return;
    }
    int dd, xx, yy;
    ExGCD(b, a % b, &dd, &xx, &yy);
    *d = dd, *x = yy, *y = xx - a / b * yy;
}
int main()
{
    int a, b, d, x, y;
    scanf("%d%d", &a, &b);
    ExGCD(a, b, &d, &x, &y);
    printf("%d = %d*(%d) + %d*(%d)", d, a, x, b, y);
    return 0;
}

示例代码 2

G 编撰幻想乡缘起

难度 考点
4~5 字符串处理，库函数的使用

题意解析

首先我们回忆一下字符串的本质：C 语言中并没有单独的字符串类型，所谓的字符串本质是是一个以空字符 \0 为结尾的一维 char 类型数组。注意这个空字符，当你将一维字符数组当做字符串来处理时，这个字符串会在第一个空字符处『结束』。例如你可以尝试运行以下代码：

char buf[10] = {'1', '2', '3', '\0', '4', '5', '6'};
printf("%s", buf);

输出为：

123

因为 %s 将 bufbufbuf 以字符串形式处理，因此只会输出第一个 \0 之前的内容，使用 %s 读取字符串时则会自动加上一个 \0 。

类似的， gets 读取字符串时会将读取到的内容最后的 \n 替换为 \0 来保证字符串的完整性（若没有 \n 而是读到了文件末尾则是添加一个 \0）

同理，字符串操作函数，包括但不限于 strlen ， strcpy ， strcat ， strstr 都是同样的道理，只会处理第一个 \0 之前的内容。而 strcpy ， strcat 这些函数也会在操作结果的字符串的对应位置上补上一个 \0 。

那么由以上内容我们就能写出各个操作对应的代码。

对于操作 111 拼接，直接调用 strcat 函数即可。

strcat(str1, str2);

对于操作 222 截断，我们在截断位置增加一个终止符即可。

str1[i] = '\0';

对于操作 333 插入，我们将原字符串拆成两段，将其与给出的字符串依次拼接即可。

strcpy(tmp, str1 + i + 1); // 将后半段复制给tmp
str1[i + 1] = '\0';        // 截断字符串，使其仅留下前半段
strcat(str1, str2);        // 拼接插入字符串
strcat(str1, tmp);         // 拼接后半段

对于操作 444 删除，其本质和操作 333 一致，如下：

strcpy(tmp, str1 + j + 1); // 将后半段复制给tmp
str1[i] = '\0';            // 截断字符串，使其仅留下前半段
strcat(str1, tmp);         // 拼接后半段

需要注意的是， strcpy 和 strcat 的 dst 和 src 不能重叠，否则是一个未定义行为，结果取决于编译器的实现方式。因此直接进行如下操作是错误的。

// !undefined behavior
strcpy(str1 + i, str1 + j + 1);

对于操作 555， 同样本质与操作 333 一致，如下

strcpy(tmp, str1 + j + 1); // 将后半段复制给tmp
str1[i] = '\0';            // 截断字符串，使其仅留下前半段
strcat(str1, str2);        // 拼接插入字符串
strcat(str1, tmp);         // 拼接后半段

对于操作 666，我们已经在 C6-D 中做过了，此处不再赘述。

char *p1 = strstr(str1, str2);
if (p1 == NULL) {
    puts("NULL");
} else {
    char *p2 = p1;
    while (strstr(p2 + 1, str2) != NULL) {
        p2 = strstr(p2 + 1, str2);
    }
    printf("%d %d\n", (int)(p1 - str1), (int)(p2 - str1));
}

H Cirno 的完美函数教室

难度 考点
5 记忆化递归，剪枝

问题分析

本题如果使用递归，需要使用记忆化的技巧进行剪枝，即用数组存储计算过的函数值，否则递归次数过多会导致 TLE。

当然，也可以使用递推，要注意递推的关系式和更新的顺序。

参考代码 1

#include <stdio.h>

int f[30][30][30];

int B(int a, int b, int c)
{
	if (a <= 0 || b <= 0 || c <= 0)
		return 1;
	else if (a > 25 || b > 25 || c > 25)
		return B(25, 25, 25);
	else
	{
		if (f[a][b][c]) // 若计算过，直接返回之前的计算结果，无需再递归调用
			return f[a][b][c];
		else if (a < b && b < c)
			f[a][b][c] = B(a, b, c - 1) + B(a, b - 1, c - 1) - B(a, b - 1, c);
		else
			f[a][b][c] = B(a - 1, b, c) + B(a - 1, b - 1, c) + B(a - 1, b, c - 1) - B(a - 1, b - 1, c - 1);
	}
	return f[a][b][c];
}

int main()
{
	int T, a, b, c;
	scanf("%d", &T);
	while (T--)
	{
		scanf("%d%d%d", &a, &b, &c);
		printf("%d\n", B(a, b, c));
	}
	return 0;
}

参考代码 2

#include <stdio.h>

int main() {
    int B[26][26][26];
    for (int i = 0; i < 26; ++i) {
        for (int j = 0; j < 26; ++j) {
            for (int k = 0; k < 26; ++k) {
                if (i == 0 || j == 0 || k == 0) {
                    B[i][j][k] = 1;
                } else {
                    if (i < j && j < k) {
                        B[i][j][k] = B[i][j][k - 1] + B[i][j - 1][k - 1] - B[i][j - 1][k];
                    } else {
                        B[i][j][k] = B[i - 1][j][k] + B[i - 1][j - 1][k] + B[i - 1][j][k - 1] - B[i - 1][j - 1][k - 1];
                    }
                }
            }
        }
    }
    int T, a, b, c;
    scanf("%d", &T);
    for (int i = 0; i < T; ++i) {
        scanf("%d%d%d", &a, &b, &c);
        if (a <= 0 || b <= 0 || c <= 0) {
            a = b = c = 0;
        } else if (a > 25 || b > 25 || c > 25) {
            a = b = c = 25;
        }
        printf("%d\n", B[a][b][c]);
    }

    return 0;
}

I Fractal

难度 考点
5~6 递归、循环、二维数组

题意分析

思路一：递归输出每一行

定义一个函数 f(k,l,m)f(k,l,m)f(k,l,m) ，表示输出 kkk 阶图案第 mmm 个部分的第 lll 行， m=0,1,2,3m = 0, 1, 2, 3m=0,1,2,3 分别代表左上、右上、左下、右下部分。

以左上部分为例，可以发现，对于 kkk 阶图案的左上部分，是由 k−1k-1k−1 阶图案的左上、右上、左下部分和空白组成的。如果 kkk 阶图案左上部分第 lll 行在左上部分的上半部分，则其等于 k−1k-1k−1 阶图案的左上和右上部分；如果第 lll 行在左上部分的下半部分，则等于 k−1k-1k−1 阶图案的左下部分和相同大小的空白。

其他情况与左上部分类似，分类讨论递归即可完成本题。

递归基本情况是 k=0k = 0k=0 时，左上、右上、左下、右下部分均为一个 1 ； kkk 阶图案的边长为 2k+12^{k+1}2k+1 ，其左上、右上、左下、右下部分的边长为 2k2^k2k ，每个部分均由四个边长为 2k−12^{k-1}2k−1 的小部分组成。

具体实现参照示例代码及注释。

思路二：递归输出每个位置

定义一个函数 f(k,i,j)f(k, i, j)f(k,i,j) ，表示输出 kkk 阶图案第 iii 行第 jjj 个位置（从第0行第0列开始记）。

  • 若 k=0k=0k=0 ，则输出 1 ；
  • 否则，若位置 (i,j)(i, j)(i,j) 在中央，即 i,j∈[2k−1,3⋅2k−1)i,j\in[2^{k-1},3\cdot2^{k-1})i,j∈[2k−1,3⋅2k−1) ，则输出空格；
  • 否则，输出 f(k−1,i′,j′)f(k-1, i', j')f(k−1,i′,j′) ，其中 i′=i mod 2k, j′=j mod 2ki'=i\bmod 2^k,\ j'=j\bmod 2^ki′=imod2k, j′=jmod2k 。

具体实现参照示例代码及注释。

该思路耗时比思路一更长，但较为简单。

思路三：循环+二维数组

由于二维数组还未学到，本思路仅作为参考。

由 kkk 阶图案生成 k+1k+1k+1 阶图案可由下列两步操作实现：

  1. 将 kkk 阶图案复制 444 份，拼在一起形成一个大正方形（即向右、向下、向右下复制平移）；
  2. 将该大正方形中心与 kkk 阶图案相同大小的区域置为空白。

依次思路， kkk 阶图案仅需要从原始图案重复执行上列操作 kkk 遍即可得到。

具体实现参照示例代码及注释。

补充

学过二维数组后似乎思路三要更加的简单一些，但是这种规律并不具有普遍性，换一个分形图案就无法用相同的规律生成了，而递归的思路却是类似的。此外思路三的空间消耗非常大，虽然本题不会爆内存，但是推荐大家写程序时考虑占用的空间大小。

#include <stdio.h>
void f(int k, int l, int m)
{
	if(!k) printf("1"); //基本情况，k=0，此时l一定为1，无论m是0,1,2,3，均输出一个1
	else if(l < (1 << (k - 1))) //如果l小于2^(k-1)，即该行在左上或右上或左下或右下部分的上半部分
	{
		switch(m) //根据m的值进行分类讨论
		{
			case 2: //左下部分的上半部分
				f(k - 1, l, 0); //输出k-1阶的左上部分的第l行
				for(int i = 0; i < (1 << (k - 1)); ++i) printf(" "); //输出相同长度的空格
				break;
			case 3: //右下部分的上半部分
				for(int i = 0; i < (1 << (k - 1)); ++i) printf(" "); //输出相同长度的空格
				f(k - 1, l, 1); //输出k-1阶的右上部分的第l行
				break;
			default: //左上或右上部分的上半部分
				f(k - 1, l, 0); //输出k-1阶的左上部分的第l行
				f(k - 1, l, 1); //输出k-1阶的右上部分的第l行
				break;
		}
	}
	else //l不小于2^(k-1)，说明该行在左上或右上或左下或右下部分的下半部分

		l -= (1 << (k - 1)); //更新l的值为k-1阶左上或右上或左下或右下部分对应的行数，即自减2^(k-1)

				break;
			default: //左下或右下部分的下半部分

思路三：循环+二维数组

#include <stdio.h>
int a[2048][2048] = {{1, 1}, {1, 1}}; //初始为四个1
int main()
{
	int K;
	scanf("%d", &K);
	for(int k = 1; k <= K; ++k) //K遍操作
	{
		for(int i = 0; i < (1 << k); ++i)
			for(int j = 0; j < (1 << k); ++j)
				a[i + (1 << k)][j + (1 << k)] = a[i + (1 << k)][j] = a[i][j + (1 << k)] = a[i][j]; //向右、下、右下复制平移
		for(int i = 0; i < (1 << k); ++i)
			for(int j = 0; j < (1 << k); ++j)
				a[i + (1 << (k - 1))][j + (1 << (k - 1))] = 0; //中心部分置零
	}
	for(int i = 0; i < (1 << (K + 1)); ++i)
    {
		for(int j = 0; j <= (1 << (K + 1)); ++j)
			printf("%c", a[i][j] ? '1' : ' '); //输出，若为1则输出1，若为0则输出空格
        printf("\n"); //每行之间要换行
    }
	return 0;
}

J 观察折线图

难度 考点
6 二分答案，计算几何

题目分析

设输入的 nnn 个点的坐标分别为 (x0,y0),(x1,y1),⋯ ,(xn−1,yn−1)(x_0,y_0),(x_1,y_1),\cdots,(x_{n-1},y_{n-1})(x0​,y0​),(x1​,y1​),⋯,(xn−1​,yn−1​)。

一个由 nnn 个点连成的折线一共有 n−1n-1n−1 条线段，设第 iii 条线段的方程为 y=aix+biy=a_ix+b_iy=ai​x+bi​（由 xi>xi−1x_i>x_{i-1}xi​>xi−1​ 知斜率 aia_iai​ 一定存在）。

则有 ai=yi−yi−1xi−xi−1,bi=yi−aixi,i∈{1,2,⋯ ,n−1}a_i=\dfrac{y_i-y_{i-1}}{x_i-x_{i-1}},b_i=y_i-a_ix_i,i\in \lbrace1,2,\cdots,n-1\rbraceai​=xi​−xi−1​yi​−yi−1​​,bi​=yi​−ai​xi​,i∈{1,2,⋯,n−1}。

我们要寻找一个点，使得该点不在折线的任意一段线段所在直线的下方。

不难发现，对于最小的满足条件的点的纵坐标 yansy_{ans}yans​，当 y>yansy>y_{ans}y>yans​ 时，一定也存在纵坐标为 yyy 且满足条件的点；当 y<yansy<y_{ans}y<yans​ 时，一定不存在纵坐标为 yyy 且满足条件的点。因此我们可以考虑二分查找该点的纵坐标 yyy。

对于一个确定的 yyy，若要使点 (x,y)(x,y)(x,y) 不在第 iii 条线段所在直线的下方，则要求 y≥aix+biy\ge a_ix+b_iy≥ai​x+bi​​，分以下三种情况进行讨论：

  1. 若 ai<0a_i<0ai​<0，则要求 x≥y−biaix\ge \dfrac{y-b_i}{a_i}x≥ai​y−bi​​；
  2. 若 ai>0a_i>0ai​>0，则要求 x≤y−biaix\le \dfrac{y-b_i}{a_i}x≤ai​y−bi​​；
  3. 若 ai=0a_i=0ai​=0，则仅需要求 y≥by\ge by≥b 即可。

对于一个确定的 yyy，遍历所有的线段，检查是否存在纵坐标为 yyy 且满足上方条件的点。

进而利用二分法，求得最小的满足条件的 yyy 即可。

示例代码

#include <stdio.h>
#include <math.h>
int n;
double x[100005], y[100005], a[100005], b[100005];
int check(double m) // 对于确定的m，判断是否存在纵坐标为m且满足条件的点
{
	double L = 0, R = 1000000; // [L,R]表示纵坐标为y时，满足条件的横坐标的范围
	for(int i = 1; i < n; i++) // 遍历n-1个线段
	{
        // 情况1，要求横坐标不小于(m-bi)/ai
		if(a[i] < 0) L = fmax(L, (m - b[i]) / a[i]);      // 更新横坐标可能的最小值L
        // 情况2，要求横坐标不大于(m-bi)/ai
		else if(a[i] > 0) R = fmin(R, (m - b[i]) / a[i]); // 更新横坐标可能的最大值L
        // 情况3，要求m不小于bi
		else if(m < b[i]) return 0;                       // 若m<bi，返回0表示不存在满足条件的点
	}
	return L <= R; // 如果L<=R，说明存在满足条件的点，否则说明不存在
}
int main()
{
	scanf("%d%lf%lf", &n, &x[0], &y[0]); // 读入n和第一个点
	for(int i = 1; i < n; i++) // 读入接下来n-1个点，并计算n-1条线段的斜率a和截距b
	{
		scanf("%lf%lf", &x[i], &y[i]);
		a[i] = (y[i] - y[i - 1]) / (x[i] - x[i - 1]);
		b[i] = y[i] - a[i] * x[i];
	}
    // 二分查找
	double l = 0, r = 1e10; // 由数据范围可知，y最小为0，最大不超过10的10次幂。
	while(r - l > 1e-5)
	{
		double m = (l + r) / 2;
		if(check(m)) r = m;
		else l = m;
	}
	printf("%.3f", l);
	return 0;
}
